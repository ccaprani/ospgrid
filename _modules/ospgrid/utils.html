[docs]def save_figs(filename, figs=None):
    """
    Saves passed figs, or all open figs to PDF
    """

    pp = PdfPages(filename)

    if figs is None:
        figs = [plt.figure(n) for n in plt.get_fignums()]

    for fig in figs:
        fig.savefig(pp, format="pdf")

    pp.close()
